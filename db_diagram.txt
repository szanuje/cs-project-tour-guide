// Creating tables
Table users as U {
  id int [pk, increment] // auto-increment
  username string
  password string
}

Table destinations as D {
  id int [pk, increment]
  name string
  description string
  places_id int
  hotels_id int
}

Table places as P {
  id int [pk, increment]
  name string
  description string
  address_id int
}

Table hotels as H {
  id int [pk, increment]
  name string
  rating string
  price decimal
  address_id int
}

Table addresses as A {
  id int [pk, increment]
  country string
  city string
  street string
  house_number string
  postal_code string
  lat decimal
  lng decimal
}


Ref : destinations.places_id > places.id
Ref : destinations.hotels_id > hotels.id

Ref : places.address_id > addresses.id
Ref : hotels.address_id > addresses.id
